91  template <typename T, typename Tcompare, typename Ttransform>
92  std::shared_ptr<wassail::result>
93  check(const json &j, const json::json_pointer &key, Tcompare cmp,
94  const T &reference_value, Ttransform transform) {
95  std::shared_ptr<wassail::result> r = make_result(j);
96  r->brief = wassail::format(fmt_str.brief, static_cast<std::string>(key),
97  value(reference_value));
98 
99  try {
100  T val = transform(j.at(key).get<T>());
101  if (cmp(val, reference_value)) {
102  r->issue = result::issue_t::NO;
103  r->priority = result::priority_t::INFO;
104  r->detail = wassail::format(fmt_str.detail_no, value(val),
105  value(reference_value));
106  }
107  else {
108  r->issue = result::issue_t::YES;
109  r->priority = result::priority_t::WARNING;
110  r->detail = wassail::format(fmt_str.detail_yes, value(val),
111  value(reference_value));
112  }
113  }
114  catch (std::exception &e) {
115  logger(log_level::warn, e.what() + std::string(": ") + j.dump());
116  r->issue = result::issue_t::MAYBE;
117  r->detail = wassail::format(fmt_str.detail_maybe, e.what(), j.dump());
118  }
119 
120  return r;
121  }

188  template <typename Titerator, typename T, typename Tcompare>
189  std::shared_ptr<wassail::result>
190  check(const Titerator &begin, const Titerator &end,
191  const json::json_pointer &key, Tcompare cmp,
192  const T &reference_value) {
193  static_assert(
194  std::is_same<typename std::iterator_traits<Titerator>::value_type,
195  json>::value,
196  "iterator type must be json");
197 
198  /* overall result, everything will be a child of this result */
199  auto r = make_result();
200 
201  /* TODO: make this format string configurable */
202  auto count = std::distance(begin, end);
203  r->brief = wassail::format(
204  "Comparing {1} values at '{0}' to reference value of '{2}'",
205  static_cast<std::string>(key), count, value(reference_value));
206 
207  auto child = [&](auto &r, const json j, const json::json_pointer &key,
208  Tcompare cmp, const T &reference_value) {
209  auto r_ = this->check(j, key, cmp, reference_value);
210  r->add_child(r_);
211  };
212 
213  std::for_each(begin, end,
214  std::bind(child, r, std::placeholders::_1, key, cmp,
215  reference_value));
216 
217  /* set the issue and priority levels to the maximum child */
218  r->issue = r->max_issue();
219  r->priority = r->max_priority();
220 
221  return r;
222  }
